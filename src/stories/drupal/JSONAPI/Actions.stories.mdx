Accept: application/vnd.api+json
Content-Type: application/vnd.api+json
X-CSRF-Token: {{csrfToken}}

{
  "data": {
    "type": "flagging--favorite",
    "attributes": {
      "entity_type": "node",
      "entity_id": "1302",
      "global": false
    },
    "relationships": {
      "flagged_entity": {
        "data": {
          "type": "node--question",
          "id": "8aae1766-38d7-4690-af3a-77b099e56e85"
